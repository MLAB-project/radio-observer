181  inline void push(const T *items, int count)
182  {
183  // If pushing more items that the buffer can hold,
184  // clear the buffer and copy the last n items to the
185  // buffer.
186  if (count >= capacity_) {
187  clear();
188  int offset = count - capacity_;
189  for (int i = 0; i < capacity_; i++) {
190  items_[i] = items[offset + i];
191  }
192  return;
193  }
194 
195  // Otherwise, start copying the pushed items to the
196  // buffer at the head.
197  int offset = head_ - items_;
198  for (int i = 0; i < count; i++) {
199  items_[offset] = items[i];
200  offset = (offset + 1) % capacity_;
201  }
202  head_ = items_ + offset;
203  size_ += count;
204  if (size_ > capacity_) size_ = capacity_;
205  }
